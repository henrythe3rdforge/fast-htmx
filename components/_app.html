#!/usr/bin/env python3 """{{ name }} - Flask + HTMX""" import os, sys, atexit, signal from pathlib import Path from flask import Flask, render_template, request, jsonify app = Flask(__name__) SERVICE = "{{ name }}" PID_FILE = Path(__file__).parent.parent / "services.pid" def register_pid(): pid, port = os.getpid(), os.environ.get("PORT", "5000") entries = [l for l in PID_FILE.read_text().splitlines() if l and not l.startswith(f"{SERVICE}:")] if PID_FILE.exists() else [] entries.append(f"{SERVICE}:{pid}:{port}") PID_FILE.write_text("\n".join(entries) + "\n") def unregister_pid(): if not PID_FILE.exists(): return entries = [l for l in PID_FILE.read_text().splitlines() if l and not l.startswith(f"{SERVICE}:")] PID_FILE.write_text("\n".join(entries) + "\n") if entries else PID_FILE.unlink(missing_ok=True) def sig_handler(s, f): unregister_pid(); sys.exit(0) register_pid() atexit.register(unregister_pid) signal.signal(signal.SIGTERM, sig_handler) signal.signal(signal.SIGINT, sig_handler) @app.route("/") def index(): return render_template("index.html") @app.route("/about") def about(): return render_template("about.html") @app.route("/contact") def contact(): return render_template("contact.html") @app.route("/contact", methods=["POST"]) def contact_submit(): return render_template("partials/contact_success.html", name=request.form.get("name", "")) @app.route("/api/click", methods=["POST"]) def api_click(): return render_template("partials/click_response.html") @app.route("/api/counter", methods=["POST"]) def api_counter(): c = int(request.form.get("count", 0)) + 1 return f'{c}' @app.route("/health") def health(): return jsonify(status="ok", service=SERVICE, pid=os.getpid()) if __name__ == "__main__": app.run(host="0.0.0.0", port=int(os.environ.get("PORT", 5000)), debug=True)
